77 int doOperation(TokenStackPtr s, PostfixListPtr postfixList, TokenPtr token)
78 {
79  // Kontrola typu operátoru a jeho další zpracování
80  if (token == NULL)
81  {
82  // Hack pro jednoduché vyprázdnění zásobníku
83  return untilLeftPar(s, postfixList);
84  }
85  else if (token->type == OPEN_BRACKET)
86  {
87  // Jedná se o otevírací závorku, provedeme uložení na zásobník
88  return TokenStack_push(s, token);
89  }
90  else if (token->type == CLOSE_BRACKET)
91  {
92  // Jedná se o zavírací závorku, provedeme volání funkce pro
93  // zpracování této situace
94  return untilLeftPar(s, postfixList);
95  }
96  else
97  {
98  // Jedná se o některý z operátorů, provedeme jeho zpracování
99  TokenPtr stack_token;
100  int stack_result;
101  int stack_priority;
102 
103  int token_priority = getTokenPriority(token);
104 
105  while(TokenStack_isEmpty(s) == false)
106  {
107  // Získání operátoru na vrcholu zásobníku
108  stack_token = TokenStack_top(s);
109 
110  // Přiřazení priorit daným operátorům
111  stack_priority = getTokenPriority(stack_token);
112 
113  // Pokud je operátorem na vrcholu zásobníku otevírací závorka
114  // pak není problém pokračovat
115  if (stack_priority == -1)
116  {
117  break;
118  }
119  // Pokud je operátorem na vrcholu zásobníku operátor stejné
120  // nebo vyšší priority, je třeba jej přepsat do řetězce
121  else if (stack_priority >= token_priority)
122  {
123  // Operátor z vrcholu zásobníku musí být vypsán
124  PostfixList_insertOperator(postfixList, stack_token);
125  }
126  else
127  {
128  // Operátor na vrcholu zásobníku má nižší prioritu, než
129  // náš operátor, můžeme jej dát na stack
130  break;
131  }
132 
133  stack_result = TokenStack_pop(s);
134  if (stack_result != NO_ERROR)
135  {
136  return INTERNAL_ERROR;
137  }
138  }
139 
140  stack_result = TokenStack_push(s, token);
141  if (stack_result != NO_ERROR)
142  {
143  return INTERNAL_ERROR;
144  }
145  }
